공통 구현 패턴

1. Client Component + Server Action

클라이언트 (폼)

"use client";
import { useTransition } from "react";
import { useRouter } from "next/navigation";
import { createClient } from "@/lib/actions/client";

export function ClientForm() {
  const [isPending, startTransition] = useTransition();
  const router = useRouter();

  function onSubmit(formData: FormData) {
    startTransition(async () => {
      try {
        await createClient(Object.fromEntries(formData));
        router.refresh();
      } catch (e) {
        setError(e.message);
      }
    });
  }
  // ...
}

서버 (액션)

"use server";
export async function createClient(input) {
  await requireRole("ADMIN");
  const parsed = createClientSchema.parse(input);
  const client = await prisma.client.create({...});
  await logAudit({...});
  revalidatePath("/admin/clients");
  return client;
}

2. 필터 → URLSearchParams

// ListFilter.tsx
const params = new URLSearchParams(searchParams.toString());
if (q) params.set("q", q); else params.delete("q");
router.push(`?${params.toString()}`);

// page.tsx (Server Component)
export default async function Page({ searchParams }) {
  const clients = await listClients({
    q: searchParams.q,
    type: searchParams.type,
  });
  return <ClientTable items={clients} />;
}

3. 직렬화 (Server → Client)

주의 — Date · Decimal 은 Server Component 에서 Client Component 로 그대로 전달 불가. 반드시 직렬화:
  • Date → date.toISOString() (문자열)
  • Decimal → Number(decimal) 또는 decimal.toString()
이후 Client 에서 필요 시 new Date(iso) 로 복원.
