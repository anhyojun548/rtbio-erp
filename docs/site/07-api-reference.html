"use server";

export async function createThing(input) {
  const user = await requireRole("ADMIN");
  const parsed = createThingSchema.parse(input);
  const thing = await prisma.thing.create({...});
  await logAudit({
    action: "THING_CREATE",
    resource: `Thing:${thing.id}`,

  });
  return thing;
}

세션 게이트 (lib/session.ts)

모든 액션은 아래 두 헬퍼 중 하나로 진입을 제한합니다.

requireRole(...roles)

// 하나 이상의 역할을 허용
const user = await requireRole(
  "TENANT_OWNER", "ADMIN", "EXEC"
);
if (!user) redirect("/403");

내부/운영자 포털용. 실패 시 /403 리다이렉트.

requireClient()

// role=CLIENT AND clientId 필수
const user = await requireClient();
// 모든 쿼리에 clientId 필터 강제
const orders = await prisma.order.findMany({
  where: { clientId: user.clientId },
});

거래처 포털 전용. clientId 로 행-레벨 격리.

에러 처리

원칙 — 모든 액션은 성공 시 데이터를 반환하고, 실패 시 한글 메시지를 포함한 Error 를 throw. 클라이언트는 try/catch 로 받아 에러 배너 표시. Prisma 코드(P2002/P2003/P2025)는 반드시 한글 메시지로 폴백.
Prisma 코드의미표준 한글 메시지 예
P2002고유 제약 위반이미 등록된 코드입니다.
P2003외래키 제약 위반참조되는 데이터가 없습니다.
P2025레코드 없음대상을 찾을 수 없습니다.

동시성 패턴

1. Advisory Lock (번호 채번)

// order.ts · invoice.ts
await prisma.$executeRaw`
  SELECT pg_advisory_xact_lock(${lockKey})
`;
const seq = await prisma.counter.update(...);

사용처: ORD-YYYYMMDD-NNN · INV-YYYYMMDD-NNN 채번.

2. SELECT FOR UPDATE (재고)

// inventory.ts · order.ts (CONFIRM)
await prisma.$queryRaw`
  SELECT physical_stock, available_stock
  FROM product_size
  WHERE id = ${sizeId}
  FOR UPDATE
`;
// 이후 UPDATE ...

사용처: 재고 입고/조정/예약/차감. 트랜잭션 필수.

3. Transaction (복합 연산)

await prisma.$transaction(async (tx) => {
  await tx.order.update({...});
  for (const item of items) {
    await tx.productSize.update({
      data: { availableStock: { decrement: qty }}
    });
    await tx.inventoryLog.create({...});
  }
});

4. Upsert (동시 재계산)

// ClosingLedger.recomputeLedger
await prisma.closingLedger.upsert({
  where: { clientId_closingMonth: {...} },
  create: {...},
  update: {...}
});

(clientId, closingMonth) unique 로 동시성 직렬화.
